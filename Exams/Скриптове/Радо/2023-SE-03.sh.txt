#!/bin/bash

if [ ${#} -ne 1 ]; then
        echo "One argument expected"
        exit 1
fi

if [ ! -d "${1}" ]; then
        echo "Dir not found"
        exit 2
fi

stopwords=$(mktemp)
total_files=$(find "${1}" -type f | wc -l)
half_files=$(echo "$total_files/2" | bc )

 while read -r file; do
        tr -d '[:punct:]' < "${file}" | tr ' ' '\n' | while read -r word; do
                if grep -w -q "${word}" "${stopwords}"; then
                        count=$(grep -w "${word}" "${stopwords}" | cut -d ' ' -f 2)
                        count=$(( count+1 ))
                        sed -i -E "s/${word} [0-9]+/${word} ${count}/"  "$stopwords"
                else
                        echo "${word} 1" >> "$stopwords"
                fi
        done
done < <(find "${1}" -type f -name "*.txt")

sed -i "2/d" "${stopwords}"
sed -i "1/d" "${stopwords}"

i=0
while read stopword count ; do
        echo "${count}"
        if [ "$count" -gt "$half_files" ]; then
                echo "${stopword}"
                i=$(( i+1 ))
        fi
        if [ $i -eq 10 ];then
                break;
        fi
done < <(sort -nr -k 2 -t ' ' "${stopwords}")

rm "${stopwords}"
